\problem{15}
\begin{enumerate}[label=\alph*.]
\item
  \begin{align*}
    P(A~\text{and}~B) &= \frac{987}{1118} * \frac{987}{1118} \\
    P(A~\text{and}~B) &= 0.779
  \end{align*}
  The events are independent.
\item
  \begin{align*}
    P(A|B) &= \frac{987}{1118} * \frac{986}{1117} \\
    P(A|B) &= 0.779
  \end{align*}
  The events are \textbf{not} independent.
\end{enumerate}
